>Sum of odd numbers

__・自分の回答__
>select n * n * n as res from nums

__・別の方の回答__

```
SELECT POWER(n, 3)::int as res from nums;
```

```
select (n^3)::int as res
from nums
```

```
CREATE OR REPLACE FUNCTION rowSumOddNumbers (n INTEGER) 
  RETURNS INTEGER AS $$
DECLARE
   counter INTEGER := n ; 
   total   INTEGER := 0 ; 
   x       INTEGER := (n * (n - 1) + 1) ;
BEGIN

  IF (n < 1) THEN
    RETURN 0 ;
  END IF; 
  
  WHILE 0 < n LOOP
    SELECT total + x into total;
    
    n := n - 1 ; 
    x := x + 2;
  END LOOP ; 
  
  RETURN total ;
END ; $$
LANGUAGE PLPGSQL;

-- Call function for every N in nums.
SELECT res.*
    FROM nums,
    LATERAL rowSumOddNumbers(nums.n) res;
```
